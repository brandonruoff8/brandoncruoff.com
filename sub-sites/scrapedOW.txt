        $(document).ready(function() {
            var comparekey = 'compare-players';

            var name = 'pippinish33#1441';
            var region = 'global';
            var platform = '3';

            if (localStorage !== undefined){
                loadSettings();

                $('#compare-box').on('click', '.compare-toggle', function() {
                    var currArr = getArray();

                    //not in hte list
                    if (currArr.filter(function (e) { return e.name == name; }).length == 0) {
                        currArr.push({ platform: platform, region: region, name: name });
                        localStorage.setItem(comparekey, JSON.stringify(currArr));

                        loadSettings();
                    } else {
                        currArr = currArr.filter(function (e) { return e.name != name; });

                        localStorage.setItem(comparekey, JSON.stringify(currArr));

                        loadSettings();
                    }

                    ga('send', 'event', 'Compare Toggle', { 'page': '/profile' });

                });



                function loadSettings() {
                    $('#compare-box').empty();

                    var arr = getArray();

                    var isThisPlayerSelected = arr.filter(function (e) { return e.name == name; }).length > 0;

                    if (isThisPlayerSelected) {
                        //highlight compare button + X button
                        $('#compare-box').html('<div id=""compare"" class=""btn-group""><a href=""javascript:void(0);"" class=""btn btn-primary compare-toggle"">Compare</a><a id=""do-compare"" href=""/compare"" class=""btn btn-default""><i class=""fa fa-arrow-right""></i></a></div>');
                    } else {
                        $('#compare-box').html('<div id=""compare"" class=""btn-group""><a href=""javascript:void(0);"" class=""btn btn-default compare-toggle"">Compare</a></div>');
                    }
                }

                function getArray() {
                    var existingCompare = localStorage.getItem(comparekey) || '';
                    var arr = existingCompare != '' ? JSON.parse(existingCompare) : [];

                    return arr;
                }
            }

        });

        $(document).ready(function() {
            var comparekey = 'compare-players';

            var name = 'pippinish33#1441';
            var region = 'global';
            var platform = '3';

            if (localStorage !== undefined){
                loadSettings();

                $('#compare-box').on('click', '.compare-toggle', function() {
                    var currArr = getArray();

                    //not in hte list
                    if (currArr.filter(function (e) { return e.name == name; }).length == 0) {
                        currArr.push({ platform: platform, region: region, name: name });
                        localStorage.setItem(comparekey, JSON.stringify(currArr));

                        loadSettings();
                    } else {
                        currArr = currArr.filter(function (e) { return e.name != name; });

                        localStorage.setItem(comparekey, JSON.stringify(currArr));

                        loadSettings();
                    }

                    ga('send', 'event', 'Compare Toggle', { 'page': '/profile' });

                });



                function loadSettings() {
                    $('#compare-box').empty();

                    var arr = getArray();

                    var isThisPlayerSelected = arr.filter(function (e) { return e.name == name; }).length > 0;

                    if (isThisPlayerSelected) {
                        //highlight compare button + X button
                        $('#compare-box').html('<div id=""compare"" class=""btn-group""><a href=""javascript:void(0);"" class=""btn btn-primary compare-toggle"">Compare</a><a id=""do-compare"" href=""/compare"" class=""btn btn-default""><i class=""fa fa-arrow-right""></i></a></div>');
                    } else {
                        $('#compare-box').html('<div id=""compare"" class=""btn-group""><a href=""javascript:void(0);"" class=""btn btn-default compare-toggle"">Compare</a></div>');
                    }
                }

                function getArray() {
                    var existingCompare = localStorage.getItem(comparekey) || '';
                    var arr = existingCompare != '' ? JSON.parse(existingCompare) : [];

                    return arr;
                }
            }

        });
